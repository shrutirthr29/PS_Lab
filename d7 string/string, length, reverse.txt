#include <stdio.h>
#include<string.h>
int main()
{
    char name[20];
    
    //scanf("%s", name);
    gets(name);
    //printf("%s", name);
    puts(name);
    
    
    /***********/
    //string length
    int i=0;
    while(name[i]!='\0'){
        i++;
    }
    printf("%d\n",i);
    /****************/
    
    //string reverse
    i=i-1;
    int j=0;
    while(j<i){
    int t= name[i];
    name[i]= name[j];
    name[j]=t;
    j++;
    i--;
    }
    puts(name);
    
    
    /************/
    
    
    return 0;
}
